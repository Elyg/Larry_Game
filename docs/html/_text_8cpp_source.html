29 // OpenGL needs textures to be in powers of two, this function will get the
30 // nearest power of two to the current value passed in
31 //---------------------------------------------------------------------------
32 unsigned int nearestPowerOfTwo ( unsigned int _num )
33 {
34  unsigned int j, k;
35  (j = _num & 0xFFFF0000) || (j = _num);
36  (k = j & 0xFF00FF00) || (k = j);
37  (j = k & 0xF0F0F0F0) || (j = k);
38  (k = j & 0xCCCCCCCC) || (k = j);
39  (j = k & 0xAAAAAAAA) || (j = k);
40  return j << 1;
41 }

232 void Text::renderText( float _x, float _y, const std::string &text ) const
233 {
234  // make sure we are in texture unit 0 as this is what the
235  // shader expects
236  glActiveTexture(GL_TEXTURE0);
237  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
238  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
239 
240  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
241  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
242  // grab an instance of the shader manager
243  ngl::ShaderLib *shader=ngl::ShaderLib::instance();
244  // use the built in text rendering shader
245  (*shader)["nglTextShader"]->use();
246  // the y pos will always be the same so set it once for each
247  // string we are rendering
248  shader->setRegisteredUniform1f("ypos",_y);
249  // now enable blending and disable depth sorting so the font renders
250  // correctly
251  glEnable(GL_BLEND);
252  glDisable(GL_DEPTH_TEST);
253  glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
254  // now loop for each of the char and draw our billboard
255  unsigned int textLength=text.length();
256 
257  for (unsigned int i = 0; i < textLength; ++i)
258  {
259  // set the shader x position this will change each time
260  // we render a glyph by the width of the char
261  shader->setRegisteredUniform1f("xpos",_x);
262  // so find the FontChar data for our current char
263 // FontChar f = m_characters[text[i].toAscii()];
264 // FontChar f = m_characters[text[i]];
265  std::map <char,FontChar>::const_iterator currentchar=m_characters.find(text[i]);
266  // make sure we have a valid shader
267  if(currentchar!=m_characters.end())
268  {
269  FontChar f=currentchar->second;
270  // bind the pre-generated texture
271  glBindTexture(GL_TEXTURE_2D, f.textureID);
272  // bind the vao
273  f.vao->bind();
274  // draw
275  f.vao->draw();
276  // now unbind the vao
277  f.vao->unbind();
278  // finally move to the next glyph x position by incrementing
279  // by the width of the char just drawn
280  _x+=f.width;
281 
282  }
283  }
284  // finally disable the blend and re-enable depth sort
285  glDisable(GL_BLEND);
286  glEnable(GL_DEPTH_TEST);
287 
288 }
