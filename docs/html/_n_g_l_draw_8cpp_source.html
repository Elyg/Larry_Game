320 void NGLDraw::makeNoise(float m_freq, float m_scale)
321 {
322  // ----------------------------- Noise Texture ---------------------------------//
323  // set the a and b for freq and scale, we pass these in to the function
324  // store 4 noise values in a single 2D texture. store perlin noise with 1 octave
325  // in the 1st component(red channel), two octaves in the green channel, 3 blue, 4 alpha
326  const static int width_noise_tex = 512;
327  const static int height_noise_tex = 512;
328 
329  GLubyte *data = new GLubyte[ width_noise_tex * height_noise_tex * 4 ];
330  float xFactor = 1.0f / (width_noise_tex - 1);
331  float yFactor = 1.0f / (height_noise_tex - 1);
332  //animate texture along x axis
333  //float incrementFreqPos = increaseLowThreshold();
334 
335  for( int row = 0; row < height_noise_tex; row++)
336  {
337  for(int col = 0; col < width_noise_tex; col++)
338  {
339  float x = xFactor * col;
340  float y = yFactor * row;
341  float sum = 0.0f;
342  float freq = m_freq;
343  float scale = m_scale;
344 
345  // Computer sum for each octave of noise
346  for( int oct = 0; oct < 4; oct++)
347  {
348  glm::vec2 p(x * freq, y * freq);
349  // use glm library perlin noise, returns float between -1, 1
350  float val = glm::perlin(p) / scale;
351 
352  sum += val;
353  float result = (sum + 1.0f) / 2.0f;
354  // Store in texture buffer
355  data[((row * width_noise_tex + col) * 4) + oct] = (GLubyte) (result * 255.0f);
356  //std::cout << "data noise tex: " << result << std::endl;
357  freq *= 2.0f; // Double the freq for each octave
358  scale *= m_scale; // Next power of b
359  }
360  }
361  }
362 
363  glActiveTexture(GL_TEXTURE1);
364  glBindTexture(GL_TEXTURE_2D, m_TextureID[1]);
365 
366  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
367  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_NEAREST_MIPMAP_NEAREST);
368 
369  glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA, width_noise_tex, height_noise_tex, 0, GL_RGBA, GL_UNSIGNED_BYTE, data);
370  glGenerateMipmap(GL_TEXTURE_2D);
